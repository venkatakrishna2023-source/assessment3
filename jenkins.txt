pipeline {
    agent any

    tools {
        // Must match the name defined in 'Manage Jenkins' -> 'Global Tool Configuration'
        maven 'Maven' 
    }

    stages {
        stage('Checkout') {
            steps {
                // Jenkins automatically checks out from SCM if configured in the job
                checkout scm
            }
        }

        stage('Build') {
            steps {
                // Compiles code and packages the JAR, skipping tests for speed
                sh 'mvn clean package -DskipTests'
            }
        }

        stage('Test') {
            steps {
                // Runs JUnit tests
                sh 'mvn test'
            }
            post {
                always {
                    // Archives test results to show them in the Jenkins UI
                    junit 'target/surefire-reports/*.xml'
                }
            }
        }
    }
}
